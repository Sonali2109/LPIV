#include <iostream>
#include <vector>
#include <omp.h>
#include <iomanip> // For setprecision

using namespace std;

class DFSGraph {
    int V;
    vector<vector<int>> adj;

public:
    DFSGraph(int V) : V(V), adj(V) {}

    void addEdge(int u, int v) {
        adj[u].push_back(v);
        adj[v].push_back(u);
    }

    void parallelDFS(int start) {
        vector<bool> visited(V, false);
        vector<int> stack;

        stack.push_back(start);
        
        cout << "Parallel DFS Traversal: ";
        
        // --- START TIMING ---
        double start_time = omp_get_wtime();

        while (!stack.empty()) {
            int u = stack.back();
            stack.pop_back();

            if (visited[u]) continue;

            visited[u] = true;
            cout << u << " ";

            #pragma omp parallel for
            for (int i = 0; i < (int)adj[u].size(); i++) {
                int v = adj[u][i];
                if (!visited[v]) {
                    #pragma omp critical
                    {
                        stack.push_back(v);
                    }
                }
            }
        }
        
        // --- END TIMING ---
        double end_time = omp_get_wtime();
        
        cout << endl;
        cout << fixed << setprecision(6);
        cout << "DFS Execution Time: " << (end_time - start_time) << " seconds" << endl;
    }
};

int main() {
    int V = 6;
    DFSGraph g(V);
    g.addEdge(0, 1);
    g.addEdge(0, 2);
    g.addEdge(1, 3);
    g.addEdge(1, 4);
    g.addEdge(2, 5);

    g.parallelDFS(0);
    return 0;
}

/*
Time Complexity (TC)Sequential: $O(V + E)$Parallel: This is harder to speed up because DFS is inherently sequential (you must finish one node to find the next deep node).The code provided parallelizes neighbor discovery, so if a node has a very high degree (many neighbors), you get a minor speedup, but the overall TC remains $O(V+E)$.Space Complexity (SC)$O(V)$: To store the visited array and the stack. In the worst case (a skewed tree), the stack can grow to size $V$.
*/
