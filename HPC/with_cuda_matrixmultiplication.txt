#include <iostream>
using namespace std;

// ---------- VECTOR ADDITION KERNEL ----------
__global__ void vectorAdd(int *A, int *B, int *C, int n) {
    int i = threadIdx.x;
    if(i < n)
        C[i] = A[i] + B[i];
}

// ---------- MATRIX MULTIPLICATION KERNEL ----------
__global__ void matrixMul(int A[10][10], int B[10][10], int C[10][10], int n) {
    int row = threadIdx.y;
    int col = threadIdx.x;

    int sum = 0;
    for(int k = 0; k < n; k++)
        sum += A[row][k] * B[k][col];

    C[row][col] = sum;
}

// ---------- MAIN ----------
int main() {
    int n;
    cout << "Enter size: ";
    cin >> n;

    // -------- VECTOR ADDITION --------
    int A[10], B[10], C[10];

    cout << "Enter vector A:\n";
    for(int i=0;i<n;i++) cin >> A[i];

    cout << "Enter vector B:\n";
    for(int i=0;i<n;i++) cin >> B[i];

    vectorAdd<<<1,n>>>(A,B,C,n);

    cout << "\nVector Addition Result:\n";
    for(int i=0;i<n;i++) cout << C[i] << " ";

    // -------- MATRIX MULTIPLICATION --------
    int M1[10][10], M2[10][10], M3[10][10];

    cout << "\nEnter Matrix A:\n";
    for(int i=0;i<n;i++)
        for(int j=0;j<n;j++)
            cin >> M1[i][j];

    cout << "Enter Matrix B:\n";
    for(int i=0;i<n;i++)
        for(int j=0;j<n;j++)
            cin >> M2[i][j];

    dim3 threads(n,n);
    matrixMul<<<1,threads>>>(M1,M2,M3,n);

    cout << "\nMatrix Multiplication Result:\n";
    for(int i=0;i<n;i++) {
        for(int j=0;j<n;j++)
            cout << M3[i][j] << " ";
        cout << endl;
    }

    return 0;
}
