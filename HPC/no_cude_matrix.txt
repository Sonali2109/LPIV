#include <iostream>
using namespace std;

// -------- VECTOR ADDITION --------
void vectorAdd(int A[], int B[], int C[], int n) {
    for(int i = 0; i < n; i++)
        C[i] = A[i] + B[i];
}

// -------- MATRIX MULTIPLICATION --------
void matrixMul(int A[10][10], int B[10][10], int C[10][10], int n) {
    for(int i = 0; i < n; i++) {
        for(int j = 0; j < n; j++) {
            C[i][j] = 0;
            for(int k = 0; k < n; k++) {
                C[i][j] += A[i][k] * B[k][j];
            }
        }
    }
}

int main() {
    int n;
    cout << "Enter size: ";
    cin >> n;

    // Vector
    int A[10], B[10], C[10];

    cout << "Enter vector A:\n";
    for(int i=0;i<n;i++) cin >> A[i];

    cout << "Enter vector B:\n";
    for(int i=0;i<n;i++) cin >> B[i];

    vectorAdd(A,B,C,n);

    cout << "\nVector Addition:\n";
    for(int i=0;i<n;i++) cout << C[i] << " ";

    // Matrix
    int M1[10][10], M2[10][10], M3[10][10];

    cout << "\nEnter Matrix A:\n";
    for(int i=0;i<n;i++)
        for(int j=0;j<n;j++)
            cin >> M1[i][j];

    cout << "Enter Matrix B:\n";
    for(int i=0;i<n;i++)
        for(int j=0;j<n;j++)
            cin >> M2[i][j];

    matrixMul(M1,M2,M3,n);

    cout << "\nMatrix Multiplication:\n";
    for(int i=0;i<n;i++) {
        for(int j=0;j<n;j++)
            cout << M3[i][j] << " ";
        cout << endl;
    }

    return 0;
}
