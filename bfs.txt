#include <iostream>
#include <vector>
#include <queue>
#include <omp.h>
#include <iomanip> 

using namespace std;

class BFSGraph {
    int V;
    vector<vector<int>> adj;

public:
    BFSGraph(int V) : V(V), adj(V) {}

    void addEdge(int u, int v) {
        adj[u].push_back(v);
        adj[v].push_back(u);
    }

    void parallelBFS(int start) {
        vector<bool> visited(V, false);
        queue<int> q;

        visited[start] = true;
        q.push(start);

        cout << "Parallel BFS Traversal: ";

        // --- START TIMING ---
        double start_time = omp_get_wtime();

        while (!q.empty()) {
            int current_level_size = q.size();

            #pragma omp parallel for
            for (int i = 0; i < current_level_size; i++) {
                int u;
                #pragma omp critical
                {
                    if (!q.empty()) {
                        u = q.front();
                        q.pop();
                        cout << u << " ";
                    }
                }

                for (int v : adj[u]) {
                    if (!visited[v]) {
                        #pragma omp critical
                        {
                            if (!visited[v]) {
                                visited[v] = true;
                                q.push(v);
                            }
                        }
                    }
                }
            }
        }
        
        // --- END TIMING ---
        double end_time = omp_get_wtime();

        cout << endl;
        cout << fixed << setprecision(6);
        cout << "BFS Execution Time: " << (end_time - start_time) << " seconds" << endl;
    }
};

int main() {
    int V = 6;
    BFSGraph g(V);
    g.addEdge(0, 1);
    g.addEdge(0, 2);
    g.addEdge(1, 3);
    g.addEdge(1, 4);
    g.addEdge(2, 5);

    g.parallelBFS(0);
    return 0;
}

/* Time Complexity (TC)Sequential: $O(V + E)$, where $V$ is vertices and $E$ is edges.Parallel: Theoretically $O(D + \frac{V+E}{P})$, where $D$ is the diameter of the graph (number of levels) and $P$ is the number of processors.Note: In the worst case (a graph that is just one long line), the parallel version doesn't save much time because $D \approx V$.Space Complexity (SC)$O(V)$: To store the visited array and the queue which, in the worst case, holds all vertices at a single level.


Here is the working of BFS summarized in 7 concise points:

Initialize Queue: Create a FIFO (First-In, First-Out) queue and a boolean visited array to track nodes.

Start Node: Mark the starting node as visited and push it into the queue to begin traversal.

Dequeue Node: Remove the front node from the queue and set it as the current node for processing.

Visit Neighbors: Identify all adjacent nodes of the current node using the graph's adjacency list or matrix.

Check Status: For each neighbor, check the visited array to see if it has been encountered before.

Enqueue New: If a neighbor is unvisited, mark it as visited and add it to the back of the queue.

Loop & Terminate: Repeat the process until the queue is empty, ensuring all reachable nodes are explored level-by-level.
*/
